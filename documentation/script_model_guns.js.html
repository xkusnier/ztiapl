/**

Represents a rifle in the game.

@class

@constructor

@name rifle

@property {number} damage - The damage dealt by the rifle.

@property {number} AbleToShoot - The ability of the rifle to shoot.

@property {number} delay - The delay time for the rifle to shoot again.

@property {Audio} audio - The sound played when the rifle is shot.

@property {Audio} audio2 - The sound played when there are no bullets left in the rifle.

@property {number} MainCursorBullets - The maximum number of bullets the rifle can hold.

@property {number} cursorBullets - The current number of bullets the rifle has.

@property {Image} cursorImage - The image of the cursor used when aiming with the rifle.

@property {number} CursorSize - The size of the cursor image.

@property {number} cursorWidth - The width of the cursor image.

@property {number} cursorHeight - The height of the cursor image.

@property {number} x - The x-coordinate of the rifle's position on the canvas.

@property {number} y - The y-coordinate of the rifle's position on the canvas.

@draw

draws object

@special

special function
*/

class rifle {
  constructor() {
    this.damage = 1;
    this.AbleToShoot = 1;
    this.delay = 80;
    this.audio = new Audio("music/RifleShot.mp3");
    this.audio2 = new Audio("music/NoBullets.mp3");
    this.MainCursorBullets = 3;
    this.cursorBullets = 3;
    this.cursorImage = new Image();
    this.cursorImage.src = "img/crossRifle.png";
    this.CursorSize = 3;
    this.cursorWidth = 150 / this.CursorSize; // Width of the cursor image
    this.cursorHeight = 150 / this.CursorSize; // Height of the cursor image

    // Attach event listener to track mouse movement
    window.addEventListener("mousemove", (e) => {
      this.x = e.clientX - this.cursorWidth / 2;
      this.y = e.clientY - this.cursorHeight / 2;
    });
  }
  /**

  Draws the cursor image at the current mouse position on the canvas.
  @method
  */
  draw() {
    // Draw the cursor image at the current mouse position
    ctx.drawImage(
      this.cursorImage,
      this.x - 100 / this.CursorSize,
      this.y - 100 / this.CursorSize,
      this.cursorWidth + 100 / this.CursorSize,
      this.cursorHeight + 100 / this.CursorSize //50 + 200/3
    );
  }
  /**

  Reloads the rifle, setting the current number of bullets to the maximum.
  @method
  */
  special() {
    this.cursorBullets = this.MainCursorBullets;
  }
}
